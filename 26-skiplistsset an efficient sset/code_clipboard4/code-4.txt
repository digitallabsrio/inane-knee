import random
from utils import new_array
from base import BaseSet

class SkiplistSSet(BaseSet):
    # Node class
    class Node(object):
        def __init__(self, x, h):
            self.x = x
            self.next = new_array(h+1)
        # Function for node height
        def height(self):
            return len(self.next) - 1
    # Function to create new node
    def _new_node(self, x, h):
        return SkiplistSSet.Node(x, h)
        
    def __init__(self, iterable=[]):
        self._initialize()
        self.add_all(iterable)
    # Initialize function
    def _initialize(self):
        self.h = 0
        self.n = 0
        self.sentinel = self._new_node(None, 32)
        self.stack = new_array(self.sentinel.height()+1)
    # Function to reset the skiplist
    def clear(self):
        self._initialize()
    # Function to find pred node
    def find_pred_node(self, x):
        u = self.sentinel
        r = self.h
        while r >= 0:
            while u.next[r] is not None and u.next[r].x < x:
                u = u.next[r]  # go right in list r
            r -= 1  # go down into list r-1
        return u
    # Function to find node using the find_pred_node()
    def find(self, x):
        u = self.find_pred_node(x)
        if u.next[0] is None: return None
        return u.next[0].x
    # Function to add value
    def add(self, x):
        u = self.sentinel
        r = self.h
        while r >= 0:
            while u.next[r] is not None and u.next[r].x < x:
                u = u.next[r]
            if u.next[r] is not None and u.next[r].x == x: return False
            self.stack[r] = u
            r -= 1
        w = self._new_node(x, self.pick_height())
        while self.h < w.height():
            self.h += 1
            self.stack[self.h] = self.sentinel   # height increased
        for i in range(len(w.next)):
            w.next[i] = self.stack[i].next[i]
            self.stack[i].next[i] = w
        self.n += 1
        return True
    # Function to remove value
    def remove(self, x):
        removed = False
        u = self.sentinel
        r = self.h
        while r >= 0:
            while u.next[r] is not None and u.next[r].x < x:
                u = u.next[r]
            if u.next[r] is not None and u.next[r].x == x:
                removed = True
                u.next[r] = u.next[r].next[r]
                if u == self.sentinel and u.next[r] is None:
                    self.h -= 1 # height has decreased
            r -= 1
        if removed: self.n -= 1
        return removed    
    # Iterator function
    def __iter__(self):
        u = self.sentinel.next[0]
        while u is not None:
            yield u.x
            u = u.next[0]
    # Function to find length using number of nodes created      
    def __len__(self):
        return self.n
    # Function to determine the height
    def pick_height(self):
        z = random.getrandbits(32)
        k = 0
        while z & 1:
            k += 1
            z = z // 2
        return k

if __name__ == "__main__":
    # Initialize variables
    n = 1000
    # Creating object
    sl = SkiplistSSet()
    # Adding values
    print("Adding " + str(n) + " elements")
    for i in range(n):
        sl.add(2 * i)
    # Searching values
    print("Searching")
    for i in range(2 * n):
        x = sl.find(i)
        assert i > 2 * (n - 1) or abs(x - i) <= 1
    print("Searching (sequential)")
    for i in range(2 * n):
        x = sl.find(i)
        assert i > 2 * (n - 1) or abs(x - i) <= 1
    print("Searching (random)")
    r = random.Random()
    for i in range(2 * n):
        j = r.randint(0, 2 * n - 1)
        x = sl.find(j)
        assert j > 2 * (n - 1) or abs(x - j) <= 1
    print("Searching (backwards)")
    for i in range(2 * n, -1, -1):
        x = sl.find(i)
        assert i > 2 * (n - 1) or abs(x - i) <= 1
    # Removing values
    print("Removing")
    for i in range(n // 2):
        sl.remove(4 * i)
    # Verifying the values
    print("Verifying")
    for i in range(2 * n):
        x = sl.find(i)
        assert i > 2 * (n - 1) or abs(x - i) <= 3
    print("Done - size() = " + str(sl.size()))
